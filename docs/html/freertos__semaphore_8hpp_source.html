107public:

163 return xSemaphoreTake(m_semaphore, ticks_to_wait);

346template <typename SemaphoreAllocator> class mutex {
347 SemaphoreAllocator m_allocator;
348 SemaphoreHandle_t m_semaphore;
349 uint8_t m_locked : 1;
350
351public:
356 mutex() : m_allocator{}, m_semaphore{nullptr}, m_locked{false} {
357 m_semaphore = m_allocator.create_mutex();
358 configASSERT(m_semaphore);
359 }
360 mutex(const mutex &) = delete;
361 mutex(mutex &&src) = delete;
367 ~mutex(void) {
368 if (m_semaphore) {
369 vSemaphoreDelete(m_semaphore);
370 }
371 }
372
373 mutex &operator=(const mutex &) = delete;
374 mutex &operator=(mutex &&src) = delete;
375
382 BaseType_t unlock() {
383 auto rc = xSemaphoreGive(m_semaphore);
384 if (rc) {
385 m_locked = false;
386 }
387 return rc;
388 }
397 BaseType_t unlock_isr(BaseType_t &high_priority_task_woken) {
398 auto rc = xSemaphoreGiveFromISR(m_semaphore, &high_priority_task_woken);
399 if (rc) {
400 m_locked = false;
401 }
402 return rc;
403 }
411 BaseType_t lock(const TickType_t ticks_to_wait = portMAX_DELAY) {
412 auto rc = xSemaphoreTake(m_semaphore, ticks_to_wait);
413 if (rc) {
414 m_locked = true;
415 }
416 return rc;
417 }
426 BaseType_t lock_isr(BaseType_t &high_priority_task_woken) {
427 auto rc = xSemaphoreTakeFromISR(m_semaphore, &high_priority_task_woken);
428 if (rc) {
429 m_locked = true;
430 }
431 return rc;
432 }
439 template <typename Rep, typename Period>
440 BaseType_t lock(const std::chrono::duration<Rep, Period> &timeout) {
441 return lock(
442 std::chrono::duration_cast<std::chrono::milliseconds>(timeout).count());
443 }
450 BaseType_t try_lock() {
451 auto rc = xSemaphoreTake(m_semaphore, 0);
452 if (rc) {
453 m_locked = true;
454 }
455 return rc;
456 }
462 bool locked(void) const { return m_locked; }
463};
