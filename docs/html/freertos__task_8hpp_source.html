173 static void task_exec(void *context) {
174 auto pThis = static_cast<task *>(context);
175 assert(nullptr != pThis);
176#if INCLUDE_vTaskSuspend
177 if (pThis->m_start_suspended) {
178 pThis->suspend();
179 }
180#endif
181 pThis->m_taskRoutine();

549template <typename TaskAllocator> class periodic_task {
550 std::chrono::milliseconds m_period;
551 task_routine_t m_on_start;
552 task_routine_t m_on_stop;
553 task_routine_t m_periodic_routine;
554 task<TaskAllocator> m_task;
555
556 void run() {
557 m_on_start();
558 while (is_running()) {
559 if (0 != m_period.count()) {
560#if configUSE_TASK_NOTIFICATIONS
561 uint32_t notification_value = 0;
562 m_task.notify_wait(0, 0, notification_value, m_period);
563#else
564 delay(m_period);
565#endif
566 }
567 m_periodic_routine();
568 }
569 m_on_stop();
570 }
571
572public:
586 template <typename Rep, typename Period>
587 periodic_task(const char *name, UBaseType_t priority,
588 task_routine_t &&on_start, task_routine_t &&on_stop,
589 task_routine_t &&periodic_routine,
590 const std::chrono::duration<Rep, Period> &period,
591 bool start_suspended = true)
592 : m_period{std::chrono::duration_cast<std::chrono::milliseconds>(period)},
593 m_on_start{std::move(on_start)}, m_on_stop{std::move(on_stop)},
594 m_periodic_routine{std::move(periodic_routine)},
595 m_task{name, priority, [this]() { run(); }, start_suspended} {}
