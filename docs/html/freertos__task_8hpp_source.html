111 auto pThis = static_cast<task *>(context);

113#if INCLUDE_vTaskSuspend

452template <typename TaskAllocator> class periodic_task {
453 std::chrono::milliseconds m_period;
454 task_routine_t m_on_start;
455 task_routine_t m_on_stop;
456 task_routine_t m_periodic_routine;
457 task<TaskAllocator> m_task;
458
459 void run() {
460 m_on_start();
461 while (is_running()) {
462 if (0 != m_period.count()) {
463#if configUSE_TASK_NOTIFICATIONS
464 uint32_t notification_value;
465 m_task.notify_wait(0, 0, notification_value, m_period);
466#else
467 delay(m_period);
468#endif
469 }
470 m_periodic_routine();
471 }
472 m_on_stop();
473 }
474
475public:
489 template <typename Rep, typename Period>
490 periodic_task(const char *name, UBaseType_t priority,
491 task_routine_t &&on_start, task_routine_t &&on_stop,
492 task_routine_t &&periodic_routine,
493 const std::chrono::duration<Rep, Period> &period,
494 bool start_suspended = true)
495 : m_period{std::chrono::duration_cast<std::chrono::milliseconds>(period)},
496 m_on_start{on_start}, m_on_stop{on_stop},
497 m_periodic_routine{periodic_routine},
498 m_task{name, priority, [this]() { run(); }, start_suspended} {}
