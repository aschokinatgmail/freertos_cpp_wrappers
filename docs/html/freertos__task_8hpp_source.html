574 // LCOV_EXCL_START - Internal periodic task run method called by FreeRTOS
575 // kernel
576 void run() {
577 m_on_start();
578 while (is_running()) {
579 if (0 != m_period.count()) {
580#if configUSE_TASK_NOTIFICATIONS
581 uint32_t notification_value = 0;
582 m_task.notify_wait(0, 0, notification_value, m_period);
583#else
584 delay(m_period);
585#endif
586 }
587 m_periodic_routine();
588 }
589 m_on_stop();
590 }
591 // LCOV_EXCL_STOP
592
593public:

613 : m_period{std::chrono::duration_cast<std::chrono::milliseconds>(period)},

615 m_periodic_routine{std::move(periodic_routine)},
616 m_task{name, priority, [this]() { run(); }, start_suspended} {}
