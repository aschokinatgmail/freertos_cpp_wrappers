100 explicit queue(const char *registred_name = nullptr)
101 : m_allocator{}, m_queue{nullptr} {
102 m_queue = m_allocator.create();
103 configASSERT(m_queue);
104 if (registred_name != nullptr) {
105 vQueueAddToRegistry(m_queue, registred_name);
106 }
107 }
108 queue(const queue &) = delete;
109 queue(queue &&src) = delete;
110 ~queue(void) {
111 if (m_queue) {
112 auto n = pcQueueGetName(m_queue);
113 if (n) {
114 vQueueUnregisterQueue(m_queue);
115 }
116 vQueueDelete(m_queue);
117 }
118 }
