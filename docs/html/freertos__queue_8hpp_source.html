191public:
198 explicit queue(const char *registred_name = nullptr)
199 : m_queue{m_allocator.create()} {
200 configASSERT(m_queue);
201 if (registred_name != nullptr) {
202 vQueueAddToRegistry(m_queue, registred_name);
203 }
204 }
205 queue(const queue &) = delete;
206 queue(queue &&src) = delete;
207 ~queue(void) {
208 if (m_queue) {
209 auto n = pcQueueGetName(m_queue);
210 if (n) {
211 vQueueUnregisterQueue(m_queue);
212 }
213 vQueueDelete(m_queue);
214 }
215 }
