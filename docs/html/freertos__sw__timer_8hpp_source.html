100using timer_callback_t = function<void()>;

107template <typename SwTimerAllocator> class timer {
108 SwTimerAllocator m_allocator;
109 timer_callback_t m_callback;
110 uint8_t m_started : 1;
111 TimerHandle_t m_timer;
112
113 static void callback_wrapper(TimerHandle_t t) {
114 auto *const self = static_cast<timer *>(pvTimerGetTimerID(t));
115 configASSERT(self);
116 self->m_callback();
117 }
118
119public:
128 explicit timer(const char *name, const TickType_t period_ticks,
129 UBaseType_t auto_reload, timer_callback_t &&callback)
130 : m_callback{std::move(callback)}, m_started{false},
131 m_timer{m_allocator.create(name, period_ticks, auto_reload, this,
132 callback_wrapper)} {
133 configASSERT(m_timer);
134 }
145 template <typename Rep, typename Period>
146 explicit timer(const char *name,
147 const std::chrono::duration<Rep, Period> &period,
148 UBaseType_t auto_reload, timer_callback_t &&callback)
149 : timer{name,
150 static_cast<TickType_t>(
151 std::chrono::duration_cast<std::chrono::milliseconds>(period)
152 .count()),
153 auto_reload, std::move(callback)} {}
154 timer(const timer &) = delete;
166 timer(timer &&src) noexcept
167 : m_allocator(std::move(src.m_allocator)),
168 m_callback(std::move(src.m_callback)), m_started(src.m_started),
169 m_timer(src.m_timer) {
170 // Transfer ownership: clear the source timer handle to prevent double
171 // deletion
172 src.m_timer = nullptr;
173 src.m_started = false;
174 }
180 ~timer(void) {
181 if (m_timer) {
182 auto rc = xTimerDelete(m_timer, portMAX_DELAY);
183 if (rc == pdPASS) {
184 while (xTimerIsTimerActive(m_timer) != pdFALSE) {
185 vTaskDelay(pdMS_TO_TICKS(1));
186 }
187 }
188 }
189 }
190
191 timer &operator=(const timer &) = delete;
192 timer &operator=(timer &&src) noexcept {
193 if (this != &src) {
194 if (m_timer) {
195 xTimerDelete(m_timer, portMAX_DELAY);
196 }
197 auto rc = xTimerStop(src.m_timer, portMAX_DELAY);
198 if (rc == pdPASS) {
199 while (xTimerIsTimerActive(src.m_timer) != pdFALSE) {
200 vTaskDelay(pdMS_TO_TICKS(1));
201 }
202 auto name = pcTimerGetName(src.m_timer);
203 auto period = xTimerGetPeriod(src.m_timer);
204 auto auto_reload = uxTimerGetReloadMode(src.m_timer);
205 rc = xTimerDelete(src.m_timer, portMAX_DELAY);
206 if (rc == pdPASS) {
207 while (xTimerIsTimerActive(src.m_timer) != pdFALSE) {
208 vTaskDelay(pdMS_TO_TICKS(1));
209 }
210 src.m_timer = nullptr;
211 m_callback = std::move(src.m_callback);
212 m_timer = m_allocator.create(name, period, auto_reload, this,
213 callback_wrapper);
214 if (m_timer) {
215 if (src.m_started) {
216 rc = xTimerStart(m_timer, portMAX_DELAY);
217 if (rc == pdPASS) {
218 m_started = true;
219 }
220 }
221 }
222 }
223 }
224 }
225 return *this;
226 }

240 if (rc) {

242 }
243 return rc;
244 }
