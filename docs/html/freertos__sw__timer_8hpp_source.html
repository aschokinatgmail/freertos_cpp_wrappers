107template <typename SwTimerAllocator> class timer {
108 SwTimerAllocator m_allocator;
109 timer_callback_t m_callback;
110 uint8_t m_started : 1;

113 // LCOV_EXCL_START - Internal FreeRTOS timer callback function
114 static void callback_wrapper(TimerHandle_t t) {
115 auto *const self = static_cast<timer *>(pvTimerGetTimerID(t));
116 configASSERT(self);
117 self->m_callback();
118 }
119 // LCOV_EXCL_STOP
120
121public:
130 explicit timer(const char *name, const TickType_t period_ticks,
131 UBaseType_t auto_reload, timer_callback_t &&callback)
132 : m_callback{std::move(callback)}, m_started{false},
133 m_timer{m_allocator.create(name, period_ticks, auto_reload, this,
134 callback_wrapper)} {
135 configASSERT(m_timer);
136 }
147 template <typename Rep, typename Period>
148 explicit timer(const char *name,
149 const std::chrono::duration<Rep, Period> &period,
150 UBaseType_t auto_reload, timer_callback_t &&callback)
151 : timer{name,
152 static_cast<TickType_t>(
153 std::chrono::duration_cast<std::chrono::milliseconds>(period)
154 .count()),
155 auto_reload, std::move(callback)} {}
156 timer(const timer &) = delete;
168 timer(timer &&src) noexcept
169 : m_allocator(std::move(src.m_allocator)),
170 m_callback(std::move(src.m_callback)), m_started(src.m_started),
171 m_timer(src.m_timer) {
172 // Transfer ownership: clear the source timer handle to prevent double
173 // deletion
174 src.m_timer = nullptr;
175 src.m_started = false;
176 }
182 ~timer(void) {
183 if (m_timer) {
184 auto rc = xTimerDelete(m_timer, portMAX_DELAY);
185 if (rc == pdPASS) {
186 while (xTimerIsTimerActive(m_timer) != pdFALSE) {
187 vTaskDelay(pdMS_TO_TICKS(1));
188 }
189 }
190 }
191 }
192
193 timer &operator=(const timer &) = delete;
194 timer &operator=(timer &&src) noexcept {
195 if (this != &src) {
196 if (m_timer) {
197 xTimerDelete(m_timer, portMAX_DELAY);
198 }
199 auto rc = xTimerStop(src.m_timer, portMAX_DELAY);
200 if (rc == pdPASS) {
201 while (xTimerIsTimerActive(src.m_timer) != pdFALSE) {
202 vTaskDelay(pdMS_TO_TICKS(1));
203 }
204 auto name = pcTimerGetName(src.m_timer);
205 auto period = xTimerGetPeriod(src.m_timer);
206 auto auto_reload = uxTimerGetReloadMode(src.m_timer);
207 rc = xTimerDelete(src.m_timer, portMAX_DELAY);
208 if (rc == pdPASS) {
209 while (xTimerIsTimerActive(src.m_timer) != pdFALSE) {
210 vTaskDelay(pdMS_TO_TICKS(1));
211 }
212 src.m_timer = nullptr;
213 m_callback = std::move(src.m_callback);
214 m_timer = m_allocator.create(name, period, auto_reload, this,
215 callback_wrapper);
216 if (m_timer) {
217 if (src.m_started) {
218 rc = xTimerStart(m_timer, portMAX_DELAY);
219 if (rc == pdPASS) {
220 m_started = true;
221 }
222 }
223 }
224 }
225 }
226 }
227 return *this;
228 }

242 if (rc) {
243 m_started = true;
244 }
245 return rc;
