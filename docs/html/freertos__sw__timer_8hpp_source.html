99using timer_callback_t = function<void()>;
100
106template <typename SwTimerAllocator> class timer {
107 SwTimerAllocator m_allocator;
108 TimerHandle_t m_timer;
109 timer_callback_t m_callback;
110 uint8_t m_started : 1;
111
112 static void callback_wrapper(TimerHandle_t t) {
113 auto *const self = static_cast<timer *>(pvTimerGetTimerID(t));
114 configASSERT(self);
115 self->m_callback();
116 }
117
118public:
127 explicit timer(const char *name, const TickType_t period_ticks,
128 UBaseType_t auto_reload, timer_callback_t &&callback)
129 : m_timer{nullptr}, m_callback{callback}, m_started{false} {
130 m_timer = m_allocator.create(name, period_ticks, auto_reload, this,
131 callback_wrapper);
132 configASSERT(m_timer);
133 }
144 template <typename Rep, typename Period>
145 explicit timer(const char *name,
146 const std::chrono::duration<Rep, Period> &period,
147 UBaseType_t auto_reload, timer_callback_t &&callback)
148 : timer{name,
149 static_cast<TickType_t>(
150 std::chrono::duration_cast<std::chrono::milliseconds>(period)
151 .count()),
152 auto_reload, std::move(callback)} {}
153 timer(const timer &) = delete;
154 timer(timer &&src) = delete;
160 ~timer(void) {
161 if (m_timer) {
162 xTimerDelete(m_timer, portMAX_DELAY);
163 }
164 }
165
166 timer &operator=(const timer &) = delete;
167 timer &operator=(timer &&src) = delete;
168
176 BaseType_t start(const TickType_t ticks_to_wait = portMAX_DELAY) {
177 auto rc = xTimerStart(m_timer, ticks_to_wait);
178 if (rc) {
179 m_started = true;
180 }
181 return rc;
182 }
